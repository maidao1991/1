  <script>

    // Confetti đơn giản

    const canvas = document.getElementById("c");

    const ctx = canvas.getContext("2d");

    let W, H, pieces = [], anim = null;


    function resize(){

      const dpr = window.devicePixelRatio || 1;

      W = canvas.width = innerWidth * dpr;

      H = canvas.height = innerHeight * dpr;

      canvas.style.width = innerWidth + "px";

      canvas.style.height = innerHeight + "px";

      ctx.setTransform(1,0,0,1,0,0);

    }

    addEventListener("resize", resize);

    resize();


    function rand(a,b){ return Math.random()*(b-a)+a; }


    function burst(n=120){

      const dpr = window.devicePixelRatio || 1;

      for(let i=0;i<n;i++){

        pieces.push({

          x: rand(0, innerWidth)*dpr,

          y: rand(-20, -5)*dpr,

          r: rand(2, 5)*dpr,

          vx: rand(-1.3, 1.3)*dpr,

          vy: rand(2.2, 4.6)*dpr,

          g: rand(0.02, 0.055)*dpr,

          a: 1,

          hue: rand(0, 360)

        });

      }

      if(!anim) animate();

    }


    function animate(){

      anim = requestAnimationFrame(animate);

      ctx.clearRect(0,0,W,H);


      for(const p of pieces){

        p.x += p.vx;

        p.y += p.vy;

        p.vy += p.g;

        p.a -= 0.004;


        ctx.globalAlpha = Math.max(p.a, 0);

        ctx.fillStyle = `hsl(${p.hue}, 90%, 65%)`;

        ctx.beginPath();

        ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);

        ctx.fill();

      }


      ctx.globalAlpha = 1;

      pieces = pieces.filter(p => p.a > 0 && p.y < H + 60);


      if(pieces.length === 0){

        cancelAnimationFrame(anim);

        anim = null;

      }

    }


    const giftBtn = document.getElementById("gift");

    const msg = document.getElementById("msg");


    giftBtn.addEventListener("click", () => {

      const open = (msg.style.display === "none");

      msg.style.display = open ? "block" : "none";

      giftBtn.textContent = open ? "💝 Đóng lại" : "🎁 Mở quà";

      burst(open ? 160 : 80);

    });
